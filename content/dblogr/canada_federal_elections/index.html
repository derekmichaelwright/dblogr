Create an Animation

library(magick)
# all images are in a folder called `maps`
fnames <- list.files("maps")
mp <- image_read(paste("maps", fnames, sep = "/")) %>% image_scale("x967")
image_write(image_animate(mp, fps = 2.5), "canadian_federal_elections.gif")
